	forward_dif (a_expected_value, a_actual_value: STRING_8): INTEGER_32
			-- Compute the first dif point moving forward from first character of both strings, until reaching the end of the shorter.
		require
			has_expected: not a_expected_value.is_empty
			has_actual: not a_actual_value.is_empty
		local
			i: INTEGER_32
			l_dif_found: BOOLEAN
		do
			from
				i := 1
			until
				l_dif_found or (i > a_expected_value.count or i > a_actual_value.count)
			loop
				l_dif_found := not a_expected_value [i].is_equal (a_actual_value [i])
				i := i + 1
			end
			if l_dif_found then
				Result := i
			end
		end

	backward_dif_on_shorter_expected (a_expected_value, a_actual_value: STRING_8): INTEGER_32
			-- Compute backward dif on shorter `a_expected_value'.
		require

			has_longer: not a_actual_value.is_empty

	diffs (a_expected_value, a_actual_value: detachable ANY): STRING_8
			-- What are the diffs between `a_expected_value' and `a_actual_value'?
		local
			l_expected, l_actual: STRING_8
			l_forward_dif_location, l_backward_dif_location: INTEGER_32
		do
			create Result.make_empty
			if attached a_expected_value then
				l_expected := a_expected_value.out
			else
				create l_expected.make_empty
			end
			if attached a_actual_value then
				l_actual := a_actual_value.out
			else
				create l_actual.make_empty
			end
			check
				has_expected_and_actual: (attached l_expected) and (attached l_actual)
			end
			if l_expected.is_empty and l_actual.is_empty then
				Result := "No difs: Both strings are empty."
			elseif l_expected.is_empty and not l_actual.is_empty then
				Result := "Expected is empty, actual is not."
			elseif not l_expected.is_empty and l_actual.is_empty then
				Result := "Actual is empty, expected is not."
			elseif not l_expected.is_empty and not l_actual.is_empty then
				l_forward_dif_location := forward_dif (l_expected, l_actual)
				if l_expected.count <= l_actual.count then
					l_backward_dif_location := backward_dif_on_shorter_expected (l_expected, l_actual)
				else
					l_backward_dif_location := backward_dif_on_shorter_actual (l_expected, l_actual)
				end
			end
			if Result.is_empty then
				if l_forward_dif_location = 0 and l_backward_dif_location = 0 then
					Result := "No difs"
				elseif l_forward_dif_location = l_backward_dif_location then
					Result := "The string are different at character #"
					Result.append_string (l_forward_dif_location.out)
					Result.append_character ('%N')
				else
					Result := "Differ from character #"
					Result.append_string (l_forward_dif_location.out)
					Result.append_string (" to #")
					Result.append_string (l_backward_dif_location.out)
					Result.append_character ('%N')
					Result.append_string ("String block: ")
					if l_forward_dif_location <= l_backward_dif_location then
						Result.append_string (l_expected.substring (l_forward_dif_location, l_backward_dif_location))
					else
						Result.append_string (l_expected.substring (l_backward_dif_location, l_forward_dif_location))
					end
					Result.append_character ('%N')
				end
			end
		ensure
			not_empty_result: not Result.is_empty
		end
